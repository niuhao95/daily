# 防抖

const debounce = (fn, time) => {
  let timer
  return (...args) => {
    let _this = this
    if(timer){
      clearTimeout(timer)
    }
    timer = setTimeout(() => {
      fn.apply(_this, args)
    }, time)
  }
}

window.onresize = debounce(() => {
  console.log(123)
}, 1000)
// 立即执行
// 返回值
// 取消

# 节流

const throttle = (fn, time) => {
  let timer
  return (...args) => {
    let _this = This
    if(!timer){
      timer = setTimeout(() => {
        timer = null
        fn.apply(_this, args)
      }, time)
    }
  }
}

window.onresize = throttle(() => {
  console.log(456)
}, 1000)
