# Vue 2 API

render(h) {
    const slot = (this.$slots.default ?? []).filter(i => i.tag)

    return h('div', {
        attrs: {
            class: 'stack'
        }
    }, slot.map(child => {
        return h('div', {
            attrs: {
                class: 'mt-4'
            }
        }, [child])
    }))
}

<template>
  <Stack>
    <div>hello</div>
    <div>hello</div>
    <Stack>
        <div>hello</div>
        <div>hello</div>
        <div>hello</div>
    </Stack>
  </Stack>
</template>
